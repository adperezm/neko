138  subroutine case_init_common(C)
139  type(case_t), target, intent(inout) :: c
140  character(len=:), allocatable :: output_directory
141  integer :: lx = 0
142  logical :: scalar = .false.
143  type(file_t) :: msh_file, bdry_file, part_file
144  type(mesh_fld_t) :: msh_part, parts
145  logical :: found, logical_val
146  integer :: integer_val
147  real(kind=rp) :: real_val
148  character(len=:), allocatable :: string_val
149  real(kind=rp) :: stats_start_time, stats_output_val

151  integer :: output_dir_len

153  integer :: precision

155  !
156  ! Load mesh
157  !
158  call json_get(c%params, 'case.mesh_file', string_val)
159  msh_file = file_t(string_val)
160

162 
163  !
164  ! Load Balancing
165  !
166  call json_get_or_default(c%params, 'case.load_balancing', logical_val,&
167  .false.)
168 
169  if (pe_size .gt. 1 .and. logical_val) then
170  call neko_log%section('Load Balancing')
171  call parmetis_partmeshkway(c%msh, parts)
172  call redist_mesh(c%msh, parts)
173  call neko_log%end_section()
174  end if
175 
176  !
177  ! Time step
178  !
179  call json_get(c%params, 'case.timestep', c%dt)
180 
181  !
182  ! End time
183  !
184  call json_get(c%params, 'case.end_time', c%end_time)
185 
186  !
187  ! Initialize point_zones registry
188  !
189  call neko_point_zone_registry%init(c%params, c%msh)
190 
191  !
192  ! Setup user defined functions
193  !
194  call c%usr%init()
195  call c%usr%user_mesh_setup(c%msh)
196 
197 
198  !
199  ! Material properties
200  !
201  call c%material_properties%init(c%params, c%usr)
202 
203  !
204  ! Setup fluid scheme
205  !
206  call json_get(c%params, 'case.fluid.scheme', string_val)
207  call fluid_scheme_factory(c%fluid, trim(string_val))
208 
209  call json_get(c%params, 'case.numerics.polynomial_order', lx)
210  lx = lx + 1 ! add 1 to get poly order
211  call c%fluid%init(c%msh, lx, c%params, c%usr, c%material_properties)
212  c%fluid%chkp%tlag => c%tlag
213  c%fluid%chkp%dtlag => c%dtlag
214  select type(f => c%fluid)
215  type is(fluid_pnpn_t)
216  f%chkp%abx1 => f%abx1
217  f%chkp%abx2 => f%abx2
218  f%chkp%aby1 => f%aby1
219  f%chkp%aby2 => f%aby2
220  f%chkp%abz1 => f%abz1
221  f%chkp%abz2 => f%abz2
222  end select
223 
224 
225  !
226  ! Setup scratch registry
227  !
228  neko_scratch_registry = scratch_registry_t(c%fluid%dm_Xh, 10, 10)
229 
230  !
231  ! Setup scalar scheme
232  !
233  ! @todo no scalar factory for now, probably not needed
234  if (c%params%valid_path('case.scalar')) then
235  call json_get_or_default(c%params, 'case.scalar.enabled', scalar,&
236  .true.)
237  end if
238 
239  if (scalar) then
240  allocate(c%scalar)
241  call c%scalar%init(c%msh, c%fluid%c_Xh, c%fluid%gs_Xh, c%params, c%usr,&
242  c%material_properties)
243  call c%fluid%chkp%add_scalar(c%scalar%s)
244  c%fluid%chkp%abs1 => c%scalar%abx1
245  c%fluid%chkp%abs2 => c%scalar%abx2
246  c%fluid%chkp%slag => c%scalar%slag
247  end if
248 
249  !
250  ! Setup user defined conditions
251  !
252  if (c%params%valid_path('case.fluid.inflow_condition')) then
253  call json_get(c%params, 'case.fluid.inflow_condition.type',&
254  string_val)
255  if (trim(string_val) .eq. 'user') then
256  call c%fluid%set_usr_inflow(c%usr%fluid_user_if)
257  end if
258  end if
259 
260  ! Setup user boundary conditions for the scalar.
261  if (scalar) then
262  call c%scalar%set_user_bc(c%usr%scalar_user_bc)
263  end if
264 
265  !
266  ! Setup initial conditions
267  !
268  call json_get(c%params, 'case.fluid.initial_condition.type',&
269  string_val)
270  if (trim(string_val) .ne. 'user') then
271  call set_flow_ic(c%fluid%u, c%fluid%v, c%fluid%w, c%fluid%p, &
272  c%fluid%c_Xh, c%fluid%gs_Xh, string_val, c%params)
273  else
274  call set_flow_ic(c%fluid%u, c%fluid%v, c%fluid%w, c%fluid%p, &
275  c%fluid%c_Xh, c%fluid%gs_Xh, c%usr%fluid_user_ic, c%params)
276  end if
277 
278  if (scalar) then
279  call json_get(c%params, 'case.scalar.initial_condition.type', string_val)
280  if (trim(string_val) .ne. 'user') then
281  call set_scalar_ic(c%scalar%s, &
282  c%scalar%c_Xh, c%scalar%gs_Xh, string_val, c%params)
283  else
284  call set_scalar_ic(c%scalar%s, &
285  c%scalar%c_Xh, c%scalar%gs_Xh, c%usr%scalar_user_ic, c%params)
286  end if
287  end if
288 
289  ! Add initial conditions to BDF scheme (if present)
290  select type(f => c%fluid)
291  type is(fluid_pnpn_t)
292  call f%ulag%set(f%u)
293  call f%vlag%set(f%v)
294  call f%wlag%set(f%w)
295  end select
296 
297 
298  !
299  ! Validate that the case is properly setup for time-stepping
300  !
301  call c%fluid%validate
302 
303  if (scalar) then
304  call c%scalar%slag%set(c%scalar%s)
305  call c%scalar%validate
306  end if
307 
308  !
309  ! Set order of timestepper
310  !
311  call json_get(c%params, 'case.numerics.time_order', integer_val)
312  call c%ext_bdf%init(integer_val)
313 
314  !
315  ! Get and process output directory
316  !
317  call json_get_or_default(c%params, 'case.output_directory',&
318  output_directory, '')
319 
320  output_dir_len = len(trim(output_directory))
321  if (output_dir_len .gt. 0) then
322  if (output_directory(output_dir_len:output_dir_len) .ne. "/") then
323  output_directory = trim(output_directory)//"/"
324  if (pe_rank .eq. 0) then
325  call execute_command_line('mkdir -p '//output_directory)
326  end if
327  end if
328  end if
329 
330  !
331  ! Save boundary markings for fluid (if requested)
332  !
333  call json_get_or_default(c%params, 'case.output_boundary',&
334  logical_val, .false.)
335  if (logical_val) then
336  bdry_file = file_t(trim(output_directory)//'bdry.fld')
337  call bdry_file%write(c%fluid%bdry)
338  end if
339 
340  !
341  ! Save mesh partitions (if requested)
342  !
343  call json_get_or_default(c%params, 'case.output_partitions',&
344  logical_val, .false.)
345  if (logical_val) then
346  call mesh_field_init(msh_part, c%msh, 'MPI_Rank')
347  msh_part%data = pe_rank
348  part_file = file_t(trim(output_directory)//'partitions.vtk')
349  call part_file%write(msh_part)
350  call mesh_field_free(msh_part)
351  end if
352 
353  !
354  ! Setup output precision of the field files
355  !
356  call json_get_or_default(c%params, 'case.output_precision', string_val,&
357  'single')
358 
359  if (trim(string_val) .eq. 'double') then
360  precision = dp
361  else
362  precision = sp
363  end if
364 
365  !
366  ! Setup sampler
367  !
368  call c%s%init(c%end_time)
369  if (scalar) then
370  c%f_out = fluid_output_t(precision, c%fluid, c%scalar, &
371  path=trim(output_directory))
372  else
373  c%f_out = fluid_output_t(precision, c%fluid, &
374  path=trim(output_directory))
375  end if
376 
377  call json_get_or_default(c%params, 'case.fluid.output_control',&
378  string_val, 'org')
379 
380  if (trim(string_val) .eq. 'org') then
381  ! yes, it should be real_val below for type compatibility
382  call json_get(c%params, 'case.nsamples', real_val)
383  call c%s%add(c%f_out, real_val, 'nsamples')
384  else if (trim(string_val) .eq. 'never') then
385  ! Fix a dummy 0.0 output_value
386  call json_get_or_default(c%params, 'case.fluid.output_value', real_val, &
387  0.0_rp)
388  call c%s%add(c%f_out, 0.0_rp, string_val)
389  else
390  call json_get(c%params, 'case.fluid.output_value', real_val)
391  call c%s%add(c%f_out, real_val, string_val)
392  end if
393 
394  !
395  ! Save checkpoints (if nothing specified, default to saving at end of sim)
396  !
397  call json_get_or_default(c%params, 'case.output_checkpoints',&
398  logical_val, .true.)
399  if (logical_val) then
400  c%f_chkp = chkp_output_t(c%fluid%chkp, path=output_directory)
401  call json_get_or_default(c%params, 'case.checkpoint_control', &
402  string_val, "simulationtime")
403  call json_get_or_default(c%params, 'case.checkpoint_value', real_val,&
404  1e10_rp)
405  call c%s%add(c%f_chkp, real_val, string_val)
406  end if
407 
408  !
409  ! Setup statistics
410  !
411 
412  ! Always init, so that we can call eval in simulation.f90 with no if.
413  ! Note, don't use json_get_or_default here, because that will break the
414  ! valid_path if statement below (the path will become valid always).
415  call c%params%get('case.statistics.start_time', stats_start_time,&
416  found)
417  if (.not. found) stats_start_time = 0.0_rp
418 
419  call c%params%get('case.statistics.sampling_interval', &
420  stats_sampling_interval, found)
421  if (.not. found) stats_sampling_interval = 10
422 
423  call c%q%init(stats_start_time, stats_sampling_interval)
424 
425  found = c%params%valid_path('case.statistics')
426  if (found) then
427  call json_get_or_default(c%params, 'case.statistics.enabled',&
428  logical_val, .true.)
429  if (logical_val) then
430  call c%q%add(c%fluid%mean%u)
431  call c%q%add(c%fluid%mean%v)
432  call c%q%add(c%fluid%mean%w)
433  call c%q%add(c%fluid%mean%p)
434 
435  c%f_mf = mean_flow_output_t(c%fluid%mean, stats_start_time, &
436  path=output_directory)
437 
438  call json_get(c%params, 'case.statistics.output_control', &
439  string_val)
440  call json_get(c%params, 'case.statistics.output_value', &
441  stats_output_val)
442 
443  call c%s%add(c%f_mf, stats_output_val, string_val)
444  call c%q%add(c%fluid%stats)
445 
446  c%f_stats_output = fluid_stats_output_t(c%fluid%stats, &
447  stats_start_time, path=output_directory)
448  call c%s%add(c%f_stats_output, stats_output_val, string_val)
449  end if
450  end if
451 
452 ! if (C%params%stats_mean_sqr_flow) then
453 ! call C%q%add(C%fluid%mean_sqr%uu)
454 ! call C%q%add(C%fluid%mean_sqr%vv)
455 ! call C%q%add(C%fluid%mean_sqr%ww)
456 ! call C%q%add(C%fluid%mean_sqr%pp)
457 
458 ! if (C%params%output_mean_sqr_flow) then
459 ! C%f_msqrf = mean_sqr_flow_output_t(C%fluid%mean_sqr, &
460 ! C%params%stats_begin, &
461 ! path=output_directory)
462 ! call C%s%add(C%f_msqrf, C%params%stats_write_par, &
463 ! C%params%stats_write_control)
464 ! end if
465 ! end if
466 
467  !
468  ! Setup joblimit
469  !
470  if (c%params%valid_path('case.job_timelimit')) then
471  call json_get(c%params, 'case.job_timelimit', string_val)
472  call jobctrl_set_time_limit(string_val)
473  end if
474 
475  call neko_log%end_section()

477  end subroutine case_init_common
