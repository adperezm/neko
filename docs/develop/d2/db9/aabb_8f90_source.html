163  function get_aabb(object, padding) result(box)
164  use utils, only: neko_error
165  implicit none
166 
167  class(*), intent(in) :: object
168  real(kind=rp), intent(in), optional :: padding
169  type(aabb_t) :: box
170 
171  select type(object)
172  type is (aabb_t)
173  box = object
174  type is (tri_t)
175  box = get_aabb_triangle(object)
176 
177  class default
178  print *, "Error: get_aabb not implemented for this type"
179  stop
180  end select
181 
182  if (present(padding)) then
183  box%box_min = box%box_min - padding * box%diameter
184  box%box_max = box%box_max + padding * box%diameter
185  end if
186 
187  end function get_aabb

195  function get_aabb_triangle(triangle) result(aabb)
196  type(tri_t), intent(in) :: triangle
197  type(aabb_t) :: aabb
198 
199  real(kind=rp), dimension(3) :: box_min, box_max
200 
201  associate(pts => triangle%pts)
202  box_min = min(pts(1)%p%x, pts(2)%p%x, pts(3)%p%x)
203  box_max = max(pts(1)%p%x, pts(2)%p%x, pts(3)%p%x)
204  end associate
205 
206  call aabb%init(box_min, box_max)
207  end function get_aabb_triangle
