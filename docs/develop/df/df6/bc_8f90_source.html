278  subroutine bc_mark_zones_from_list(this, bc_zones, bc_key, bc_labels)
279  class(bc_t), intent(inout) :: this
280  class(facet_zone_t), intent(inout) :: bc_zones(:)
281  character(len=*) :: bc_key
282  character(len=20) :: bc_labels(NEKO_MSH_MAX_ZLBLS)
283  integer :: i, j, k, msh_bc_type
284 
285  msh_bc_type = 0
286  if(trim(bc_key) .eq. 'o' .or. trim(bc_key) .eq. 'on' &
287  .or. trim(bc_key) .eq. 'o+dong' .or. trim(bc_key) .eq. 'on+dong') then

298  if (trim(bc_key) .eq. trim(bc_labels(i))) then
299  call bc_mark_zone(this, bc_zones(i))
300  ! Loop across all faces in the mesh
301  do j = 1,this%msh%nelv
302  do k = 1, 2 * this%msh%gdim
303  if (this%msh%facet_type(k,j) .eq. -i) then
304  this%msh%facet_type(k,j) = msh_bc_type
305  end if
306  end do
307  end do

316  subroutine bc_finalize(this)
317  class(bc_t), target, intent(inout) :: this
318  type(tuple_i4_t), pointer :: bfp(:)
319  type(tuple_i4_t) :: bc_facet
320  integer :: facet_size, facet, el
321  integer :: i, j, k, l, msk_c
322  integer :: lx, ly, lz, n
323 
324  lx = this%Xh%lx
325  ly = this%Xh%ly
326  lz = this%Xh%lz
327 
329 
330  ! Note we assume that lx = ly = lz
331  facet_size = lx**2
332  allocate(this%msk(0:facet_size * this%marked_facet%size()))
333  allocate(this%facet(0:facet_size * this%marked_facet%size()))
334 
335  msk_c = 0
336  bfp => this%marked_facet%array()
337 
338  ! Loop through each (facet, element) id tuple
339  ! Then loop over all the nodes of the face and compute their linear index
340  ! This index goes into this%msk, whereas the corresponding face id goes into
341  ! this%facet
342  do i = 1, this%marked_facet%size()
343  bc_facet = bfp(i)
344  facet = bc_facet%x(1)
345  el = bc_facet%x(2)
346  select case (facet)
347  case (1)
348  do l = 1, lz
349  do k = 1, ly
350  msk_c = msk_c + 1
351  this%msk(msk_c) = linear_index(1,k,l,el,lx,ly,lz)
352  this%facet(msk_c) = 1
353  end do
354  end do
355  case (2)
356  do l = 1, lz
357  do k = 1, ly
358  msk_c = msk_c + 1
359  this%msk(msk_c) = linear_index(lx,k,l,el,lx,ly,lz)
360  this%facet(msk_c) = 2
361  end do
362  end do
363  case(3)
364  do l = 1, lz
365  do j = 1, lx
366  msk_c = msk_c + 1
367  this%msk(msk_c) = linear_index(j,1,l,el,lx,ly,lz)
368  this%facet(msk_c) = 3
369  end do
370  end do
371  case(4)
372  do l = 1, lz
373  do j = 1, lx
374  msk_c = msk_c + 1
375  this%msk(msk_c) = linear_index(j,ly,l,el,lx,ly,lz)
376  this%facet(msk_c) = 4
377  end do
378  end do
379  case(5)
380  do k = 1, ly
381  do j = 1, lx
382  msk_c = msk_c + 1
383  this%msk(msk_c) = linear_index(j,k,1,el,lx,ly,lz)
384  this%facet(msk_c) = 5
385  end do
386  end do
387  case(6)
388  do k = 1, ly
389  do j = 1, lx
390  msk_c = msk_c + 1
391  this%msk(msk_c) = linear_index(j,k,lz,el,lx,ly,lz)
392  this%facet(msk_c) = 6
393  end do
394  end do
395  end select
396  end do
397 
398  this%msk(0) = msk_c
399  this%facet(0) = msk_c
400 
401  if (neko_bcknd_device .eq. 1) then
402  n = facet_size * this%marked_facet%size() + 1
403  call device_map(this%msk, this%msk_d, n)
404  call device_map(this%facet, this%facet_d, n)
405 
406  call device_memcpy(this%msk, this%msk_d, n, &
407  host_to_device, sync=.false.)
408  call device_memcpy(this%facet, this%facet_d, n, &
409  host_to_device, sync=.false.)
410  end if
411 
412  end subroutine bc_finalize
