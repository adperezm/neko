79  subroutine data_streamer_init(this, coef, if_asynch)
80  class(data_streamer_t), intent(inout) :: this
81  type(coef_t), intent(inout) :: coef
82  integer, intent(in) :: if_asynch
83  integer :: nelb, nelb2, nelv, nelgv,npts,e
84 
85  !Allocate and initialize the global element number
86  allocate(this%lglel(coef%msh%nelv))
87  do e = 1, coef%msh%nelv
88  this%lglel(e) = e + coef%msh%offset_el
89  end do
90 
91  !Assign if the streaming is asynchronous
92  this%if_asynch = if_asynch
93 
94  !Assign the set up parameters
95  nelv = coef%msh%nelv
96  npts = coef%Xh%lx*coef%Xh%ly*coef%Xh%lz
97  nelgv = coef%msh%glb_nelv
98  nelb = coef%msh%offset_el
99  ! Alternative way to get nelb:
100  !nelb = elem_running_sum(nelv)
101  !nelb = nelb - nelv
102 
103 #ifdef HAVE_ADIOS2
104  call fortran_adios2_setup(npts, nelv, nelb, nelgv, &
105  nelgv, coef%dof%x, coef%dof%y, &
106  coef%dof%z, if_asynch, neko_comm)
107 #else
108  call neko_warning('Is not being built with ADIOS2 support.')
109  call neko_warning('Not able to use stream/compression functionality')
110 #endif
111 
112 
113  end subroutine data_streamer_init

138  subroutine data_streamer_stream(this, u, v, w, p, coef)
139  class(data_streamer_t), intent(inout) :: this
140  type(coef_t), intent(inout) :: coef
141  type(field_t), intent(inout) :: u
142  type(field_t), intent(inout) :: v
143  type(field_t), intent(inout) :: w
144  type(field_t), intent(inout) :: p
145  integer :: nelv, npts
146 
147  nelv = coef%msh%nelv
148  npts = coef%Xh%lx*coef%Xh%ly*coef%Xh%lz
149 
150  if (neko_bcknd_device .eq. 1) then
151  ! Move the data to the CPU to be able to write it
152  call device_memcpy(u%x, u%x_d, nelv*npts, device_to_host, sync=.false.)
153  call device_memcpy(v%x, v%x_d, nelv*npts, device_to_host, sync=.false.)
154  call device_memcpy(w%x, w%x_d, nelv*npts, device_to_host, sync=.false.)
155  call device_memcpy(p%x, p%x_d, nelv*npts, device_to_host, sync=.true.)
156  end if
157 
158 #ifdef HAVE_ADIOS2
159  call fortran_adios2_stream(this%lglel, p%x, u%x, v%x, w%x, coef%B, u%x)
160 #else
161  call neko_warning('Is not being built with ADIOS2 support.')
162  call neko_warning('Not able to use stream/compression functionality')
163 #endif
164 
165  end subroutine data_streamer_stream
