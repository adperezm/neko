206  subroutine scalar_scheme_add_bcs(this, zones, bc_labels)
207  class(scalar_scheme_t), intent(inout) :: this
208  type(facet_zone_t), intent(inout) :: zones(NEKO_MSH_MAX_ZLBLS)
209  character(len=20), intent(in) :: bc_labels(:)

211  integer :: i, j, bc_idx
212  real(kind=rp) :: dir_value, flux_value
213  logical :: bc_exists
214 
215  do i = 1, size(bc_labels)
216  bc_label = trim(bc_labels(i))
217  if (bc_label(1:1) .eq. 'd') then
218 ! The idea of this commented piece of code is to merge bcs with the same
219 ! Dirichlet value into 1 so that one has less kernel launches. Currently
220 ! segfaults, needs investigation.
221 ! bc_exists = .false.
222 ! bc_idx = 0
223 ! do j = 1, i-1
224 ! if (bc_label .eq. bc_labels(j)) then
225 ! bc_exists = .true.
226 ! bc_idx = j
227 ! end if
228 ! end do
229 
230 ! if (bc_exists) then
231 ! call this%dir_bcs(j)%mark_zone(zones(i))
232 ! else
233  this%n_dir_bcs = this%n_dir_bcs + 1
234  call this%dir_bcs(this%n_dir_bcs)%init(this%dm_Xh)
235  call this%dir_bcs(this%n_dir_bcs)%mark_zone(zones(i))
236  read(bc_label(3:), *) dir_value
237  call this%dir_bcs(this%n_dir_bcs)%set_g(dir_value)
238 ! end if
239  end if
240 
241  if (bc_label(1:1) .eq. 'n') then
242  this%n_neumann_bcs = this%n_neumann_bcs + 1
243  call this%neumann_bcs(this%n_neumann_bcs)%init(this%dm_Xh)
244  call this%neumann_bcs(this%n_neumann_bcs)%mark_zone(zones(i))
245  read(bc_label(3:), *) flux_value
246  call this%neumann_bcs(this%n_neumann_bcs)%init_neumann(flux_value,&
247  this%c_Xh)
248  end if
249 
251  if (bc_label(1:4) .eq. 'user') then
252  call this%user_bc%mark_zone(zones(i))
253  end if
254  end do
255 
256  do i = 1, this%n_dir_bcs
257  call this%dir_bcs(i)%finalize()
258  call bc_list_add(this%bclst_dirichlet, this%dir_bcs(i))
259  end do
260 
261  ! Create list with just Neumann bcs
262  call bc_list_init(this%bclst_neumann, this%n_neumann_bcs)
263  do i=1, this%n_neumann_bcs
264  call this%neumann_bcs(i)%finalize()
265  call bc_list_add(this%bclst_neumann, this%neumann_bcs(i))
266  end do

277  subroutine scalar_scheme_init(this, msh, c_Xh, gs_Xh, params, scheme, user, &
278  material_properties)
279  class(scalar_scheme_t), target, intent(inout) :: this
280  type(mesh_t), target, intent(inout) :: msh
281  type(coef_t), target, intent(inout) :: c_Xh

286  type(material_properties_t), target, intent(inout) :: material_properties
287  ! IO buffer for log output
288  character(len=LOG_SIZE) :: log_buf
289  ! Variables for retrieving json parameters

294 
295  this%u => neko_field_registry%get_field('u')
296  this%v => neko_field_registry%get_field('v')
297  this%w => neko_field_registry%get_field('w')
298 
299  call neko_log%section('Scalar')
300  call json_get(params, 'case.fluid.velocity_solver.type', solver_type)
301  call json_get(params, 'case.fluid.velocity_solver.preconditioner',&
302  solver_precon)
303  call json_get(params, 'case.fluid.velocity_solver.absolute_tolerance',&
304  solver_abstol)
305 
306  !
307  ! Material properties
308  !
309  this%rho => material_properties%rho
310  this%lambda => material_properties%lambda
311  this%cp => material_properties%cp
312 
313  write(log_buf, '(A,ES13.6)') 'rho :', this%rho
314  call neko_log%message(log_buf)
315  write(log_buf, '(A,ES13.6)') 'lambda :', this%lambda
316  call neko_log%message(log_buf)
317  write(log_buf, '(A,ES13.6)') 'cp :', this%cp
318  call neko_log%message(log_buf)
319 
320  call json_get_or_default(params, &
321  'case.fluid.velocity_solver.projection_space_size',&
322  this%projection_dim, 20)
323 
324 
325  write(log_buf, '(A, A)') 'Type : ', trim(scheme)
326  call neko_log%message(log_buf)
327  call neko_log%message('Ksp scalar : ('// trim(solver_type) // &
328  ', ' // trim(solver_precon) // ')')
329  write(log_buf, '(A,ES13.6)') ' `-abs tol :', solver_abstol
330  call neko_log%message(log_buf)
331 
332  this%Xh => this%u%Xh
333  this%dm_Xh => this%u%dof
334  this%params => params
335  this%msh => msh
336  call neko_field_registry%add_field(this%dm_Xh, 's')
337  this%s => neko_field_registry%get_field('s')
338 
339  this%gs_Xh => gs_xh
340  this%c_Xh => c_xh
341 
342 
343  !
344  ! Setup scalar boundary conditions
345  !
346  call bc_list_init(this%bclst_dirichlet)
347  call this%user_bc%init(this%dm_Xh)
348 
349  ! Read boundary types from the case file
350  allocate(this%bc_labels(neko_msh_max_zlbls))
351 
352  ! A filler value
353  this%bc_labels = "not"
354 
355  if (params%valid_path('case.scalar.boundary_types')) then
356  call json_get(params, &
357  'case.scalar.boundary_types', &
358  this%bc_labels)
359  end if
360 
361 
362  !
363  ! Setup right-hand side field.
364  !
365  allocate(this%f_Xh)
366  call this%f_Xh%init(this%dm_Xh, fld_name="scalar_rhs")
367 
368  ! Initialize the source term
369  call this%source_term%init(params, this%f_Xh, this%c_Xh, user)
370 
371  call scalar_scheme_add_bcs(this, msh%labeled_zones, this%bc_labels)
372 
373 
374  call this%user_bc%mark_zone(msh%wall)
375  call this%user_bc%mark_zone(msh%inlet)
376  call this%user_bc%mark_zone(msh%outlet)
377  call this%user_bc%mark_zone(msh%outlet_normal)
378  call this%user_bc%mark_zone(msh%sympln)
379  call this%user_bc%finalize()
380  call this%user_bc%set_coef(this%c_Xh)
381  if (this%user_bc%msk(0) .gt. 0) call bc_list_add(this%bclst_dirichlet,&
382  this%user_bc)
383 
384 
385  ! todo parameter file ksp tol should be added
386  call json_get_or_default(params, 'case.fluid.velocity_solver.max_iterations',&
387  integer_val, 800)
388  call scalar_scheme_solver_factory(this%ksp, this%dm_Xh%size(), &
389  solver_type, integer_val, solver_abstol)
390  call scalar_scheme_precon_factory(this%pc, this%ksp, &
391  this%c_Xh, this%dm_Xh, this%gs_Xh, this%bclst_dirichlet, solver_precon)
392 
393  call neko_log%end_section()
394 
395  end subroutine scalar_scheme_init
