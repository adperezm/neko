137  function tri_mesh_brute_force(mesh, p, max_distance) result(distance)
138  use tri, only: tri_t

143 
144  type(tri_mesh_t), intent(in) :: mesh
145  real(kind=rp), intent(in) :: p(3)
146  real(kind=rp), intent(in) :: max_distance
147 
148  integer :: id
149  real(kind=rp) :: distance, weighted_sign
150  real(kind=rp) :: cd, cs
151  real(kind=rp) :: tol = 1e-6_rp
152 
153  distance = 1e10_rp
154  weighted_sign = 0.0_rp
155 
156  do id = 1, mesh%nelv
157  call element_distance(mesh%el(id), p, cd, cs)
158 
159  ! Update the weighted sign, if the relative difference is small
160  if (abs(cd - distance) / distance .lt. tol) then
161  weighted_sign = weighted_sign + cs
162  else if (cd .lt. distance) then
163  weighted_sign = cs
164  end if
165 
166  distance = min(cd, distance)
167  end do
168

186  function tri_mesh_aabb_tree(tree, object_list, p, max_distance) result(distance)
187  use aabb, only: aabb_t
188  use aabb_tree, only: aabb_node_t, aabb_null_node
189  use stack, only: stack_i4_t
190  implicit none
191 
192  class(aabb_tree_t), intent(in) :: tree
193  class(tri_t), dimension(:), intent(in) :: object_list
194  real(kind=rp), dimension(3), intent(in) :: p
195  real(kind=rp), intent(in) :: max_distance
196 
197  real(kind=rp) :: distance
198  real(kind=rp) :: weighted_sign
199 
200  real(kind=rp), parameter :: tol = 1.0e-6_rp
201 
202  type(stack_i4_t) :: simple_stack
203  integer :: current_index
204 
205  type(aabb_node_t) :: current_node
206  type(aabb_t) :: current_aabb
207  integer :: current_object_index
208  real(kind=rp) :: current_distance
209  real(kind=rp) :: current_sign
210 
211  type(aabb_node_t) :: left_node
212  type(aabb_node_t) :: right_node
213 
214  type(aabb_t) :: root_box
215  type(aabb_t) :: search_box
216 
217  integer :: root_index, left_index, right_index
218  real(kind=rp) :: random_value
219 
220  ! Initialize the stack and the search box
221  call simple_stack%init(size(object_list) * 2)
222  call search_box%init(p - max_distance, p + max_distance)
223 
224  ! Check if the root node overlaps the search box, if it does, push it to
225  ! the stack and update the search box to a randomly selected object.
226  root_index = tree%get_root_index()
227  root_box = tree%get_aabb(root_index)
228 
229  if (.not. root_box%overlaps(search_box)) then
230  distance = max_distance
231  weighted_sign = 1.0_rp
232  return
233  end if
234 
235  ! Grab a random object and compute the distance to it
236  call random_number(random_value)
237  current_object_index = floor(random_value * size(object_list) + 1)
238  call element_distance(object_list(current_object_index), p, distance, weighted_sign)
239  distance = distance + object_list(current_object_index)%diameter()
240 
241  ! Update the search box to the new distance and push the root node
242  call search_box%init(p - distance, p + distance)
243  call simple_stack%push(root_index)
244 
245  ! Traverse the tree and compute the signed distance to the elements
246  do while (.not. simple_stack%is_empty())
247  current_index = simple_stack%pop()
248  if (current_index .eq. aabb_null_node) cycle
249 
250  current_node = tree%get_node(current_index)
251  current_aabb = current_node%get_aabb()
252 
253  if (current_node%is_leaf()) then
254  if (distance .lt. current_aabb%min_distance(p)) then
255  cycle
256  end if
257 
258  current_object_index = current_node%get_object_index()
259  call element_distance(object_list(current_object_index), p, current_distance, current_sign)
260 
261  ! Update the weighted sign, if the relative difference is small
262  if (abs(current_distance - distance) / distance .lt. tol) then
263  weighted_sign = weighted_sign + current_sign
264  else if (current_distance .lt. distance) then
265  weighted_sign = current_sign
266  end if
267 
268  distance = min(distance, current_distance)
269 
270  ! Update the search box to the new distance
271  if (distance .gt. current_aabb%get_diameter()) then
272  call search_box%init(p - distance, p + distance)
273  end if
274  else
275 
276  left_index = tree%get_left_index(current_index)
277  if (left_index .ne. aabb_null_node) then
278  left_node = tree%get_left_node(current_index)
279  if (left_node%aabb%overlaps(search_box)) then
280  call simple_stack%push(left_index)
281  end if
282  end if
283 
284  right_index = tree%get_right_index(current_index)
285  if (right_index .ne. aabb_null_node) then
286  right_node = tree%get_right_node(current_index)
287  if (right_node%aabb%overlaps(search_box)) then
288  call simple_stack%push(right_index)
289  end if
290  end if
291  end if
292  end do
293 
294  if (distance .gt. max_distance) then
295  distance = max_distance
296  end if
297  distance = sign(distance, weighted_sign)
298 
299  end function tri_mesh_aabb_tree

342  subroutine element_distance_triangle(triangle, p, distance, weighted_sign)
343  type(tri_t), intent(in) :: triangle
344  real(kind=rp), dimension(3), intent(in) :: p
345 
346  real(kind=rp), intent(out) :: distance
347  real(kind=rp), intent(out), optional :: weighted_sign
348 
349  real(kind=rp), dimension(3) :: v1, v2, v3
350  real(kind=rp), dimension(3) :: normal
351  real(kind=rp) :: normal_length
352  real(kind=rp) :: b1, b2, b3
353 
354  real(kind=rp), dimension(3) :: projection
355  real(kind=rp), dimension(3) :: edge
356  real(kind=rp) :: tol = 1e-10_rp
357 
358  real(kind=rp) :: face_distance
359  real(kind=rp) :: t
360 
361  ! Get vertices and the normal vector
362  v1 = triangle%pts(1)%p%x
363  v2 = triangle%pts(2)%p%x
364  v3 = triangle%pts(3)%p%x
365 
366  normal = cross(v2 - v1, v3 - v1)
367  normal_length = norm2(normal)
368 
369  if (normal_length .lt. tol) then
370  distance = huge(1.0_rp)
371  weighted_sign = 0.0_rp
372  return
373  end if
374  normal = normal / normal_length
375 
376  ! Compute Barycentric coordinates to determine if the point is inside the
377  ! triangular prism, of along an edge or by a face.
378  face_distance = dot_product(p - v1, normal)
379 
380  projection = p - normal * face_distance
381  b1 = dot_product(normal, cross(v2 - v1, projection - v1)) / normal_length
382  b2 = dot_product(normal, cross(v3 - v2, projection - v2)) / normal_length
383  b3 = dot_product(normal, cross(v1 - v3, projection - v3)) / normal_length
384 
385  if (b1 .le. tol) then
386  edge = v2 - v1
387  t = dot_product(p - v1, edge) / norm2(edge)
388  t = max(0.0_rp, min(1.0_rp, t))
389 
390  projection = v1 + t * edge
391  else if (b2 .le. tol) then
392  edge = v3 - v2
393  t = dot_product(p - v2, edge) / norm2(edge)
394  t = max(0.0_rp, min(1.0_rp, t))
395 
396  projection = v2 + t * edge
397  else if (b3 .le. tol) then
398  edge = v1 - v3
399  t = dot_product(p - v3, edge) / norm2(edge)
400  t = max(0.0_rp, min(1.0_rp, t))
401 
402  projection = v3 + t * edge
403  end if
404 
405  distance = norm2(projection - p)
406  if (present(weighted_sign)) then
407  weighted_sign = face_distance / distance
408  else
409  distance = sign(distance, face_distance)
410  end if
411 
412  end subroutine element_distance_triangle
