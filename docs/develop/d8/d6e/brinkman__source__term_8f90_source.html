157  subroutine init_boundary_mesh(this, json)
158  use file, only: file_t
159  use tri_mesh, only: tri_mesh_t
160  use device, only: device_memcpy, host_to_device
161  use filters, only: smooth_step_field, step_function_field, permeability_field
162  use signed_distance, only: signed_distance_field
163  use profiler, only: profiler_start_region, profiler_end_region
164  implicit none
165 
166  class(brinkman_source_term_t), intent(inout) :: this
167  type(json_file), intent(inout) :: json
168 
169  ! Options
170  character(len=:), allocatable :: mesh_file_name
171  character(len=:), allocatable :: distance_transform
172  character(len=:), allocatable :: filter_type
173 
174  real(kind=rp), dimension(:), allocatable :: brinkman_limits
175  real(kind=rp) :: brinkman_penalty
176 
177  type(file_t) :: mesh_file
178  type(tri_mesh_t) :: boundary_mesh
179  real(kind=rp) :: scalar
180 
181  ! ------------------------------------------------------------------------ !
182  ! Read the options for the boundary mesh
183 
184  call json_get(json, 'region.name', mesh_file_name)
185 
186  ! Settings on how to filter the design field
187  call json_get(json, 'distance_transform.type', distance_transform)
188  call json_get_or_default(json, 'filter.type', filter_type, 'none')
189 
190  ! Read the options for the permeability field
191  call json_get(json, 'brinkman.limits', brinkman_limits)
192  call json_get(json, 'brinkman.penalty', brinkman_penalty)
193 
194  if (size(brinkman_limits) .ne. 2) then
195  call neko_error('brinkman_limits must be a 2 element array of reals')
196  end if
197 
198  ! ------------------------------------------------------------------------ !
199  ! Load the immersed boundary mesh
200 
201  mesh_file = file_t(mesh_file_name)
202  call mesh_file%read(boundary_mesh)
203 
204  if (boundary_mesh%nelv .eq. 0) then
205  call neko_error('No elements in the boundary mesh')
206  end if
207 
208  ! ------------------------------------------------------------------------ !
209  ! Compute the permeability field
210 
211  ! Assign the signed distance field to all GLL points in the permeability
212  ! field. Initally we just run a brute force loop over all GLL points and
213  ! compute the signed distance function. This should be replaced with a
214  ! more efficient method, such as a tree search.
215 
216  ! Select how to transform the distance field to a design field
217  select case (distance_transform)
218  case ('smooth_step')
219  call json_get(json, 'distance_transform.value', scalar)
220 
221  call signed_distance_field(this%brinkman, boundary_mesh, scalar)
222  call smooth_step_field(this%brinkman, scalar, 0.0_rp)
223 
224  case ('step')
225 
226  call json_get(json, 'distance_transform.value', scalar)
227 
228  call signed_distance_field(this%brinkman, boundary_mesh, scalar)
229  call step_function_field(this%brinkman, scalar, 1.0_rp, 0.0_rp)
230 
231  case default
232  call neko_error('Unknown distance transform')
233  end select
234 
235  ! Run filter on the permeability field to smooth it out.
236  ! This filter should initially be the classic heaviside filter, but we wish
237  ! to alter it to be a PDE based filter to avoid MPI communication.
238  ! The "helmholtz" solver in Neko should be able to solve the PDE filter.
239 
240  select case (filter_type)
241  case ('none')
242  ! Do nothing
243  case default
244  call neko_error('Unknown filter type')
245  end select
246 
247  ! ------------------------------------------------------------------------ !
248  ! Compute the permeability field and copy to the device
249 
250  call permeability_field(this%brinkman, &
251  & brinkman_limits(1), brinkman_limits(2), brinkman_penalty)
252 
253  ! Copy the permeability field to the device
254  if (neko_bcknd_device .eq. 1) then
255  call device_memcpy(this%brinkman%x, this%brinkman%x_d, &
256  this%brinkman%dof%size(), host_to_device, .true.)
257  end if
258 
259  end subroutine init_boundary_mesh
