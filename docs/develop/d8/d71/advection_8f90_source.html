303  subroutine compute_advection_dealias(this, vx, vy, vz, fx, fy, fz, Xh, coef, n)
304  implicit none
305  class(adv_dealias_t), intent(inout) :: this
306  type(space_t), intent(inout) :: Xh
307  type(coef_t), intent(inout) :: coef
308  type(field_t), intent(inout) :: vx, vy, vz
309  integer, intent(in) :: n
310  real(kind=rp), intent(inout), dimension(n) :: fx, fy, fz
311  real(kind=rp), dimension(this%Xh_GL%lxyz) :: tx, ty, tz
312  real(kind=rp), dimension(this%Xh_GL%lxyz) :: tfx, tfy, tfz
313  real(kind=rp), dimension(this%Xh_GL%lxyz) :: vr, vs, vt
314  real(kind=rp), dimension(this%Xh_GLL%lxyz) :: tempx, tempy, tempz
315  type(c_ptr) :: fx_d, fy_d, fz_d
316  integer :: e, i, idx, nel, n_gl
317  nel = coef%msh%nelv
318  n_gl = nel * this%Xh_GL%lxyz
319  !This is extremely primitive and unoptimized on the device //Karp
320  associate(c_gl => this%coef_GL)
321  if (neko_bcknd_device .eq. 1) then
322  fx_d = device_get_ptr(fx)
323  fy_d = device_get_ptr(fy)
324  fz_d = device_get_ptr(fz)
325  call this%GLL_to_GL%map(this%tx, vx%x, nel, this%Xh_GL)
326  call this%GLL_to_GL%map(this%ty, vy%x, nel, this%Xh_GL)
327  call this%GLL_to_GL%map(this%tz, vz%x, nel, this%Xh_GL)
328 
329  call opgrad(this%vr, this%vs, this%vt, this%tx, c_gl)
330  call device_vdot3(this%tbf_d, this%vr_d, this%vs_d, this%vt_d, &
331  this%tx_d, this%ty_d, this%tz_d, n_gl)
332  call this%GLL_to_GL%map(this%temp, this%tbf, nel, this%Xh_GLL)
333  call device_sub2(fx_d, this%temp_d, n)
334 
335 
336  call opgrad(this%vr, this%vs, this%vt, this%ty, c_gl)
337  call device_vdot3(this%tbf_d, this%vr_d, this%vs_d, this%vt_d, &
338  this%tx_d, this%ty_d, this%tz_d, n_gl)
339  call this%GLL_to_GL%map(this%temp, this%tbf, nel, this%Xh_GLL)
340  call device_sub2(fy_d, this%temp_d, n)
341 
342  call opgrad(this%vr, this%vs, this%vt, this%tz, c_gl)
343  call device_vdot3(this%tbf_d, this%vr_d, this%vs_d, this%vt_d, &
344  this%tx_d, this%ty_d, this%tz_d, n_gl)
345  call this%GLL_to_GL%map(this%temp, this%tbf, nel, this%Xh_GLL)
346  call device_sub2(fz_d, this%temp_d, n)
347 
348  else if ((neko_bcknd_sx .eq. 1) .or. (neko_bcknd_xsmm .eq. 1)) then
349 
350  call this%GLL_to_GL%map(this%tx, vx%x, nel, this%Xh_GL)
351  call this%GLL_to_GL%map(this%ty, vy%x, nel, this%Xh_GL)
352  call this%GLL_to_GL%map(this%tz, vz%x, nel, this%Xh_GL)
353 
354  call opgrad(this%vr, this%vs, this%vt, this%tx, c_gl)
355  call vdot3(this%tbf, this%vr, this%vs, this%vt, &
356  this%tx, this%ty, this%tz, n_gl)
357  call this%GLL_to_GL%map(this%temp, this%tbf, nel, this%Xh_GLL)
358  call sub2(fx, this%temp, n)
359 
360 
361  call opgrad(this%vr, this%vs, this%vt, this%ty, c_gl)
362  call vdot3(this%tbf, this%vr, this%vs, this%vt, &
363  this%tx, this%ty, this%tz, n_gl)
364  call this%GLL_to_GL%map(this%temp, this%tbf, nel, this%Xh_GLL)
365  call sub2(fy, this%temp, n)
366 
367  call opgrad(this%vr, this%vs, this%vt, this%tz, c_gl)
368  call vdot3(this%tbf, this%vr, this%vs, this%vt, &
369  this%tx, this%ty, this%tz, n_gl)
370  call this%GLL_to_GL%map(this%temp, this%tbf, nel, this%Xh_GLL)
371  call sub2(fz, this%temp, n)
372 
373  else
374 
375  do e = 1, coef%msh%nelv
376  call this%GLL_to_GL%map(tx, vx%x(1,1,1,e), 1, this%Xh_GL)
377  call this%GLL_to_GL%map(ty, vy%x(1,1,1,e), 1, this%Xh_GL)
378  call this%GLL_to_GL%map(tz, vz%x(1,1,1,e), 1, this%Xh_GL)
379 
380  call opgrad(vr, vs, vt, tx, c_gl, e, e)
381  do i = 1, this%Xh_GL%lxyz
382  tfx(i) = tx(i)*vr(i) + ty(i)*vs(i) + tz(i)*vt(i)
383  end do
384 
385  call opgrad(vr, vs, vt, ty, c_gl, e, e)
386  do i = 1, this%Xh_GL%lxyz
387  tfy(i) = tx(i)*vr(i) + ty(i)*vs(i) + tz(i)*vt(i)
388  end do
389 
390  call opgrad(vr, vs, vt, tz, c_gl, e, e)
391  do i = 1, this%Xh_GL%lxyz
392  tfz(i) = tx(i)*vr(i) + ty(i)*vs(i) + tz(i)*vt(i)
393  end do
394 
395  call this%GLL_to_GL%map(tempx, tfx, 1, this%Xh_GLL)
396  call this%GLL_to_GL%map(tempy, tfy, 1, this%Xh_GLL)
397  call this%GLL_to_GL%map(tempz, tfz, 1, this%Xh_GLL)
398 
399  idx = (e-1)*this%Xh_GLL%lxyz+1
400  call sub2(fx(idx), tempx, this%Xh_GLL%lxyz)
401  call sub2(fy(idx), tempy, this%Xh_GLL%lxyz)
402  call sub2(fz(idx), tempz, this%Xh_GLL%lxyz)
403  end do
404  end if
405  end associate
406 
407  end subroutine compute_advection_dealias

420  subroutine compute_advection_no_dealias(this, vx, vy, vz, fx, fy, fz, Xh, coef, n)
421  implicit none
422  class(adv_no_dealias_t), intent(inout) :: this
423  type(space_t), intent(inout) :: Xh
424  type(coef_t), intent(inout) :: coef
425  type(field_t), intent(inout) :: vx, vy, vz
426  integer, intent(in) :: n
427  real(kind=rp), intent(inout), dimension(n) :: fx, fy, fz
428  type(c_ptr) :: fx_d, fy_d, fz_d
429 
430  if (neko_bcknd_device .eq. 1) then
431  fx_d = device_get_ptr(fx)
432  fy_d = device_get_ptr(fy)
433  fz_d = device_get_ptr(fz)
434 
435  call conv1(this%temp, vx%x, vx%x, vy%x, vz%x, xh, coef)
436  call device_subcol3 (fx_d, coef%B_d, this%temp_d, n)
437  call conv1(this%temp, vy%x, vx%x, vy%x, vz%x, xh, coef)
438  call device_subcol3 (fy_d, coef%B_d, this%temp_d, n)
439  if (coef%Xh%lz .eq. 1) then
440  call device_rzero (this%temp_d, n)
441  else
442  call conv1(this%temp, vz%x, vx%x, vy%x, vz%x, xh, coef)
443  call device_subcol3(fz_d, coef%B_d, this%temp_d, n)
444  end if
445  else
446  call conv1(this%temp, vx%x, vx%x, vy%x, vz%x, xh, coef)
447  call subcol3 (fx, coef%B, this%temp, n)
448  call conv1(this%temp, vy%x, vx%x, vy%x, vz%x, xh, coef)
449  call subcol3 (fy, coef%B, this%temp, n)
450  if (coef%Xh%lz .eq. 1) then
451  call rzero (this%temp, n)
452  else
453  call conv1(this%temp, vz%x, vx%x, vy%x, vz%x, xh, coef)
454  call subcol3(fz, coef%B, this%temp, n)
455  end if
456  end if

458  end subroutine compute_advection_no_dealias

471  subroutine compute_scalar_advection_no_dealias(this, vx, vy, vz, s, fs, Xh, &
472  coef, n)
473  implicit none
474  class(adv_no_dealias_t), intent(inout) :: this
475  type(field_t), intent(inout) :: vx, vy, vz
476  type(field_t), intent(inout) :: s
477  integer, intent(in) :: n
478  real(kind=rp), intent(inout), dimension(n) :: fs
479  type(space_t), intent(inout) :: xh
480  type(coef_t), intent(inout) :: coef
481  type(c_ptr) :: fs_d
482 
483  if (neko_bcknd_device .eq. 1) then
484  fs_d = device_get_ptr(fs)
485 
486  call conv1(this%temp, s%x, vx%x, vy%x, vz%x, xh, coef)
487  call device_subcol3 (fs_d, coef%B_d, this%temp_d, n)
488  if (coef%Xh%lz .eq. 1) then
489  call device_rzero (this%temp_d, n)
490  end if
491  else
492  ! temp will hold vx*ds/dx + vy*ds/dy + vz*ds/sz
493  call conv1(this%temp, s%x, vx%x, vy%x, vz%x, xh, coef)
494 
495  ! fs = fs - B*temp
496  call subcol3 (fs, coef%B, this%temp, n)
497  if (coef%Xh%lz .eq. 1) then
498  call rzero (this%temp, n)
499  end if
500  end if
501 
502  end subroutine compute_scalar_advection_no_dealias

515  subroutine compute_scalar_advection_dealias(this, vx, vy, vz, s, fs, Xh, &
516  coef, n)
517  implicit none
518  class(adv_dealias_t), intent(inout) :: this
519  type(field_t), intent(inout) :: vx, vy, vz
520  type(field_t), intent(inout) :: s
521  integer, intent(in) :: n
522  real(kind=rp), intent(inout), dimension(n) :: fs
523  type(space_t), intent(inout) :: xh
524  type(coef_t), intent(inout) :: coef
525  type(c_ptr) :: fs_d
526  real(kind=rp), dimension(this%Xh_GL%lxyz) :: vx_gl, vy_gl, vz_gl, s_gl
527  real(kind=rp), dimension(this%Xh_GL%lxyz) :: dsdx, dsdy, dsdz
528  real(kind=rp), dimension(this%Xh_GL%lxyz) :: f_gl
529  integer :: e, i, idx, nel, n_GL
530  real(kind=rp), dimension(this%Xh_GLL%lxyz) :: temp
531 
532  nel = coef%msh%nelv
533  n_gl = nel * this%Xh_GL%lxyz
534 
535  associate(c_gl => this%coef_GL)
536  if (neko_bcknd_device .eq. 1) then
537  fs_d = device_get_ptr(fs)
538 
539  ! Map advecting velocity onto the higher-order space
540  call this%GLL_to_GL%map(this%tx, vx%x, nel, this%Xh_GL)
541  call this%GLL_to_GL%map(this%ty, vy%x, nel, this%Xh_GL)
542  call this%GLL_to_GL%map(this%tz, vz%x, nel, this%Xh_GL)
543 
544  ! Map the scalar onto the high-order space
545  call this%GLL_to_GL%map(this%temp, s%x, nel, this%Xh_GL)
546 
547  ! Compute the scalar gradient in the high-order space
548  call opgrad(this%vr, this%vs, this%vt, this%temp, c_gl)
549 
550  ! Compute the convective term, i.e dot the velocity with the scalar grad
551  call device_vdot3(this%tbf_d, this%vr_d, this%vs_d, this%vt_d, &
552  this%tx_d, this%ty_d, this%tz_d, n_gl)
553 
554  ! Map back to the original space (we reuse this%temp)
555  call this%GLL_to_GL%map(this%temp, this%tbf, nel, this%Xh_GLL)
556 
557  ! Update the source term
558  call device_sub2(fs_d, this%temp_d, n)
559 
560  else if ((neko_bcknd_sx .eq. 1) .or. (neko_bcknd_xsmm .eq. 1)) then
561 
562  ! Map advecting velocity onto the higher-order space
563  call this%GLL_to_GL%map(this%tx, vx%x, nel, this%Xh_GL)
564  call this%GLL_to_GL%map(this%ty, vy%x, nel, this%Xh_GL)
565  call this%GLL_to_GL%map(this%tz, vz%x, nel, this%Xh_GL)
566 
567  ! Map the scalar onto the high-order space
568  call this%GLL_to_GL%map(this%temp, s%x, nel, this%Xh_GL)
569 
570  ! Compute the scalar gradient in the high-order space
571  call opgrad(this%vr, this%vs, this%vt, this%temp, c_gl)
572 
573  ! Compute the convective term, i.e dot the velocity with the scalar grad
574  call vdot3(this%tbf, this%vr, this%vs, this%vt, &
575  this%tx, this%ty, this%tz, n_gl)
576 
577  ! Map back to the original space (we reuse this%temp)

603 
604  idx = (e-1)*this%Xh_GLL%lxyz + 1
605

607  end do
608  end if
