134  subroutine scalar_pnpn_init(this, msh, coef, gs, params, user, &
135  material_properties)
136  class(scalar_pnpn_t), target, intent(inout) :: this
137  type(mesh_t), target, intent(inout) :: msh
138  type(coef_t), target, intent(inout) :: coef
139  type(gs_t), target, intent(inout) :: gs
140  type(json_file), target, intent(inout) :: params
141  type(user_t), target, intent(in) :: user
142  type(material_properties_t), intent(inout) :: material_properties
143  integer :: i
144  character(len=15), parameter :: scheme = 'Modular (Pn/Pn)'
145  ! Variables for retrieving json parameters
146  logical :: found, logical_val
147  integer :: integer_val
148 
149  call this%free()
150 
151  ! Initiliaze base type.
152  call this%scheme_init(msh, coef, gs, params, scheme, user, &
153  material_properties)
154 
155  ! Setup backend dependent Ax routines
156  call ax_helm_factory(this%ax)
157 
158  ! Setup backend dependent scalar residual routines
159  call scalar_residual_factory(this%res)
160 
161  ! Setup backend dependent summation of extrapolation scheme
162  call rhs_maker_ext_fctry(this%makeext)
163 
164  ! Setup backend depenent contributions to F from lagged BD terms
165  call rhs_maker_bdf_fctry(this%makebdf)
166 
167  ! Initialize variables specific to this plan
168  associate(xh_lx => this%Xh%lx, xh_ly => this%Xh%ly, xh_lz => this%Xh%lz, &
169  dm_xh => this%dm_Xh, nelv => this%msh%nelv)
170 
171  call this%s_res%init(dm_xh, "s_res")
172 
173  call this%abx1%init(dm_xh, "abx1")
174 
175  call this%abx2%init(dm_xh, "abx2")
176 
177  call this%ds%init(dm_xh, 'ds')
178 
179  call this%slag%init(this%s, 2)
180 
181  end associate
182 
183  ! Initialize dirichlet bcs for scalar residual
184  ! todo: look that this works
185  call this%bc_res%init(this%dm_Xh)
186  do i = 1, this%n_dir_bcs
187  call this%bc_res%mark_facets(this%dir_bcs(i)%marked_facet)
188  end do
189 
190  ! Check for user bcs
191  if (this%user_bc%msk(0) .gt. 0) then
192  call this%bc_res%mark_facets(this%user_bc%marked_facet)
193  end if
194  call this%bc_res%finalize()
195  call this%bc_res%set_g(0.0_rp)
196 
197  call bc_list_init(this%bclst_ds)
198  call bc_list_add(this%bclst_ds, this%bc_res)
199 
200 
201  ! Intialize projection space
202  if (this%projection_dim .gt. 0) then
203  call this%proj_s%init(this%dm_Xh%size(), this%projection_dim)
204  end if
205 
206  ! Add lagged term to checkpoint
207  ! @todo Init chkp object, note, adding 3 slags
208  ! call this%chkp%add_lag(this%slag, this%slag, this%slag)
209 
210  ! Uses sthe same parameter as the fluid to set dealiasing
211  call json_get(params, 'case.numerics.dealias', logical_val)
212  call params%get('case.numerics.dealiased_polynomial_order', integer_val, &
213  found)
214  if (.not. found) then
215  call json_get(params, 'case.numerics.polynomial_order', integer_val)
216  integer_val = 3.0_rp / 2.0_rp * (integer_val + 1) - 1
217  end if
218  call advection_factory(this%adv, this%c_Xh, logical_val, integer_val + 1)
219  end subroutine scalar_pnpn_init

222  subroutine scalar_pnpn_restart(this, dtlag, tlag)
223  class(scalar_pnpn_t), target, intent(inout) :: this
224  real(kind=rp) :: dtlag(10), tlag(10)
225  integer :: n
226 
227 
228  n = this%s%dof%size()
229

232  call col2(this%slag%lf(2)%x, this%c_Xh%mult, n)
233  if (neko_bcknd_device .eq. 1) then
234  call device_memcpy(this%s%x, this%s%x_d, &

241  n, host_to_device, sync=.false.)
242  call device_memcpy(this%abx2%x, this%abx2%x_d, &
243  n, host_to_device, sync=.false.)
244  end if
245 
246  call this%gs_Xh%op(this%s,gs_op_add)
247  call this%gs_Xh%op(this%slag%lf(1),gs_op_add)
248  call this%gs_Xh%op(this%slag%lf(2),gs_op_add)

289  subroutine scalar_pnpn_step(this, t, tstep, dt, ext_bdf)
290  class(scalar_pnpn_t), intent(inout) :: this
291  real(kind=rp), intent(inout) :: t
292  integer, intent(inout) :: tstep
293  real(kind=rp), intent(in) :: dt
294  type(time_scheme_controller_t), intent(inout) :: ext_bdf
295  ! Number of degrees of freedom
296  integer :: n
297  ! Linear solver results monitor
298  type(ksp_monitor_t) :: ksp_results(1)
299  character(len=LOG_SIZE) :: log_buf
300 
301  n = this%dm_Xh%size()
302 
303  call profiler_start_region('Scalar', 2)
304  associate(u => this%u, v => this%v, w => this%w, s => this%s, &
305  cp => this%cp, lambda => this%lambda, rho => this%rho, &
306  ds => this%ds, &
307  s_res =>this%s_res, &
308  ax => this%Ax, f_xh => this%f_Xh, xh => this%Xh, &
309  c_xh => this%c_Xh, dm_xh => this%dm_Xh, gs_xh => this%gs_Xh, &
310  slag => this%slag, &
311  projection_dim => this%projection_dim, &
312  msh => this%msh, res => this%res, &
313  makeext => this%makeext, makebdf => this%makebdf)
314 
315  if (neko_log%level_ .ge. neko_log_debug) then
316  write(log_buf,'(A,A,E15.7,A,E15.7,A,E15.7)') 'Scalar debug',&
317  ' l2norm s', glsc2(this%s%x,this%s%x,n),&
318  ' slag1', glsc2(this%slag%lf(1)%x,this%slag%lf(1)%x,n),&
319  ' slag2', glsc2(this%slag%lf(2)%x,this%slag%lf(2)%x,n)
320  call neko_log%message(log_buf)
321  write(log_buf,'(A,A,E15.7,A,E15.7)') 'Scalar debug2',&
322  ' l2norm abx1', glsc2(this%abx1%x,this%abx1%x,n),&
323  ' abx2', glsc2(this%abx2%x,this%abx2%x,n)
324  call neko_log%message(log_buf)
325  end if
326 
327  ! Compute the source terms
328  call this%source_term%compute(t, tstep)
329 
330  ! Pre-multiply the source terms with the mass matrix.
331  if (neko_bcknd_device .eq. 1) then
332  call device_col2(f_xh%x_d, c_xh%B_d, n)
333  else
334  call col2(f_xh%x, c_xh%B, n)
335  end if
336 
337  ! Apply Neumann boundary conditions
338  call bc_list_apply_scalar(this%bclst_neumann, this%f_Xh%x, dm_xh%size())
339 
340  ! Add the advection operators to the right-hans-side.
341  call this%adv%compute_scalar(u, v, w, s, f_xh%x, &
342  xh, this%c_Xh, dm_xh%size())
343 
344  ! At this point the RHS contains the sum of the advection operator,
345  ! Neumann boundary sources and additional source terms, evaluated using
346  ! the scalar field from the previous time-step. Now, this value is used in
347  ! the explicit time scheme to advance these terms in time.
348  call makeext%compute_scalar(this%abx1, this%abx2, f_xh%x, rho, &
349  ext_bdf%advection_coeffs, n)
350 
351  ! Add the RHS contributions coming from the BDF scheme.
352  call makebdf%compute_scalar(slag, f_xh%x, s, c_xh%B, rho, dt, &
353  ext_bdf%diffusion_coeffs, ext_bdf%ndiff, n)
354 
355  call slag%update()
356 
360  call bc_list_apply_scalar(this%bclst_dirichlet, this%s%x, this%dm_Xh%size())
361 
362  ! Compute scalar residual.
363  call profiler_start_region('Scalar residual', 20)
364  call res%compute(ax, s, s_res, f_xh, c_xh, msh, xh, lambda, rho * cp, &
365  ext_bdf%diffusion_coeffs(1), dt, &
366  dm_xh%size())
367 
368  call gs_xh%op(s_res, gs_op_add)
369 
370  ! Apply a 0-valued Dirichlet boundary conditions on the ds.
371  call bc_list_apply_scalar(this%bclst_ds, s_res%x, dm_xh%size())
372 
373  call profiler_end_region
374 
375  if (tstep .gt. 5 .and. projection_dim .gt. 0) then
376  call this%proj_s%project_on(s_res%x, c_xh, n)
377  end if
378 
379  call this%pc%update()
380  call profiler_start_region('Scalar solve', 21)
381  ksp_results(1) = this%ksp%solve(ax, ds, s_res%x, n, &
382  c_xh, this%bclst_ds, gs_xh)
383  call profiler_end_region
384 
385  if (tstep .gt. 5 .and. projection_dim .gt. 0) then
386  call this%proj_s%project_back(ds%x, ax, c_xh, &
387  this%bclst_ds, gs_xh, n)
388  end if
389 
390  ! Update the solution
391  if (neko_bcknd_device .eq. 1) then
392  call device_add2s2(s%x_d, ds%x_d, 1.0_rp, n)
393  else
394  call add2s2(s%x, ds%x, 1.0_rp, n)
395  end if
396 
397  call scalar_step_info(tstep, t, dt, ksp_results)
398 
399  end associate
400  call profiler_end_region
401  end subroutine scalar_pnpn_step
402
