194  subroutine fluid_scheme_init_common(this, msh, lx, params, scheme, user, &
195  material_properties)
196  implicit none
197  class(fluid_scheme_t), target, intent(inout) :: this
198  type(mesh_t), target, intent(inout) :: msh
199  integer, intent(inout) :: lx
200  character(len=*), intent(in) :: scheme
201  type(json_file), target, intent(inout) :: params
202  type(user_t), target, intent(in) :: user
203  type(material_properties_t), target, intent(inout) :: material_properties
204  type(dirichlet_t) :: bdry_mask
205  character(len=LOG_SIZE) :: log_buf
206  real(kind=rp), allocatable :: real_vec(:)
207  real(kind=rp) :: real_val
208  logical :: logical_val
209  character(len=:), allocatable :: string_val1, string_val2
210  ! A local pointer that is needed to make Intel happy
211 
212 
213  call neko_log%section('Fluid')
214  write(log_buf, '(A, A)') 'Type : ', trim(scheme)
215  call neko_log%message(log_buf)
216  if (lx .lt. 10) then
217  write(log_buf, '(A, I1)') 'lx : ', lx
218  else if (lx .ge. 10) then
219  write(log_buf, '(A, I2)') 'lx : ', lx
220  else
221  write(log_buf, '(A, I3)') 'lx : ', lx
222  end if
223 
224  !
225  ! Material properties
226  !
227 
228  this%rho => material_properties%rho
229  this%mu => material_properties%mu
230 
231  call neko_log%message(log_buf)
232  write(log_buf, '(A,ES13.6)') 'rho :', this%rho
233  call neko_log%message(log_buf)
234  write(log_buf, '(A,ES13.6)') 'mu :', this%mu
235 
236  call json_get(params, 'case.fluid.velocity_solver.type', string_val1)
237  call json_get(params, 'case.fluid.velocity_solver.preconditioner', &
238  string_val2)
239  call json_get(params, 'case.fluid.velocity_solver.absolute_tolerance', &
240  real_val)
241  call neko_log%message(log_buf)
242  call neko_log%message('Ksp vel. : ('// trim(string_val1) // &
243  ', ' // trim(string_val2) // ')')
244 
245  write(log_buf, '(A,ES13.6)') ' `-abs tol :', real_val
246 
247  call json_get(params, 'case.fluid.pressure_solver.type', string_val1)
248  call json_get(params, 'case.fluid.pressure_solver.preconditioner', &
249  string_val2)
250  call json_get(params, 'case.fluid.pressure_solver.absolute_tolerance', &
251  real_val)
252  call neko_log%message(log_buf)
253  call neko_log%message('Ksp prs. : ('// trim(string_val1) // &
254  ', ' // trim(string_val2) // ')')
255  write(log_buf, '(A,ES13.6)') ' `-abs tol :', real_val
256  call neko_log%message(log_buf)
257 
258  call json_get(params, 'case.numerics.dealias', logical_val)
259  write(log_buf, '(A, L1)') 'Dealias : ', logical_val
260  call neko_log%message(log_buf)
261 
262  call json_get_or_default(params, 'case.output_boundary', logical_val, &
263  .false.)
264  write(log_buf, '(A, L1)') 'Save bdry : ', logical_val
265  call neko_log%message(log_buf)
266 
267 
268  call json_get_or_default(params, &
269  'case.fluid.velocity_solver.projection_space_size',&
270  this%vel_projection_dim, 20)
271  call json_get_or_default(params, &
272  'case.fluid.pressure_solver.projection_space_size',&
273  this%pr_projection_dim, 20)
274 
275  call json_get_or_default(params, 'case.fluid.freeze', this%freeze, .false.)
276 
277  if (params%valid_path("case.fluid.flow_rate_force")) then
278  this%forced_flow_rate = .true.
279  end if
280 
281  if (msh%gdim .eq. 2) then
282  call this%Xh%init(gll, lx, lx)
283  else
284  call this%Xh%init(gll, lx, lx, lx)
285  end if
286 
287  this%dm_Xh = dofmap_t(msh, this%Xh)
288 
289  this%params => params
290 
291  this%msh => msh
292 
293  call this%gs_Xh%init(this%dm_Xh)
294 
295  call this%c_Xh%init(this%gs_Xh)
296 
297 
298  this%scratch = scratch_registry_t(this%dm_Xh, 10, 2)
299 
300  allocate(this%bc_labels(neko_msh_max_zlbls))
301  this%bc_labels = "not"
302 
303  !
304  ! Setup velocity boundary conditions
305  !
306  if (params%valid_path('case.fluid.boundary_types')) then
307  call json_get(params, &
308  'case.fluid.boundary_types', &
309  this%bc_labels)
310  end if
311 
312  call bc_list_init(this%bclst_vel)
313 
314  call this%bc_sym%init(this%dm_Xh)
315  call this%bc_sym%mark_zone(msh%sympln)
316  call this%bc_sym%mark_zones_from_list(msh%labeled_zones,&
317  'sym', this%bc_labels)
318  call this%bc_sym%finalize()
319  call this%bc_sym%init_msk(this%c_Xh)
320  call bc_list_add(this%bclst_vel, this%bc_sym)
321 
322  !
323  ! Inflow
324  !
325  if (params%valid_path('case.fluid.inflow_condition')) then
326  call json_get(params, 'case.fluid.inflow_condition.type', string_val1)
327  if (trim(string_val1) .eq. "uniform") then
328  allocate(inflow_t::this%bc_inflow)
329  else if (trim(string_val1) .eq. "blasius") then
330  allocate(blasius_t::this%bc_inflow)
331  else if (trim(string_val1) .eq. "user") then
332  allocate(usr_inflow_t::this%bc_inflow)
333  else
334  call neko_error('Invalid inflow condition '//string_val1)
335  end if
336 
337  call this%bc_inflow%init(this%dm_Xh)
338  call this%bc_inflow%mark_zone(msh%inlet)
339  call this%bc_inflow%mark_zones_from_list(msh%labeled_zones,&
340  'v', this%bc_labels)
341  call this%bc_inflow%finalize()
342  call bc_list_add(this%bclst_vel, this%bc_inflow)
343 
344  if (trim(string_val1) .eq. "uniform") then
345  call json_get(params, 'case.fluid.inflow_condition.value', real_vec)
346  call this%bc_inflow%set_inflow(real_vec)
347  else if (trim(string_val1) .eq. "blasius") then
348  select type(bc_if => this%bc_inflow)
349  type is(blasius_t)
350  call bc_if%set_coef(this%C_Xh)
351  call json_get(params, 'case.fluid.blasius.delta', real_val)
352  call json_get(params, 'case.fluid.blasius.approximation',&
353  string_val2)
354  call json_get(params, 'case.fluid.blasius.freestream_velocity',&
355  real_vec)
356 
357  call bc_if%set_inflow(real_vec)
358  call bc_if%set_params(real_val, string_val2)
359 
360  end select
361  else if (trim(string_val1) .eq. "user") then
362  select type(bc_if => this%bc_inflow)
363  type is(usr_inflow_t)
364  call bc_if%set_coef(this%C_Xh)
365  end select
366  end if
367  end if
368 
369  call this%bc_wall%init(this%dm_Xh)
370  call this%bc_wall%mark_zone(msh%wall)
371  call this%bc_wall%mark_zones_from_list(msh%labeled_zones,&
372  'w', this%bc_labels)
373  call this%bc_wall%finalize()
374  call bc_list_add(this%bclst_vel, this%bc_wall)
375 
376  call json_get_or_default(params, 'case.output_boundary', logical_val,&
377  .false.)
378  if (logical_val) then
379  call this%bdry%init(this%dm_Xh, 'bdry')
380  this%bdry = 0.0_rp
381 
382  call bdry_mask%init(this%dm_Xh)
383  call bdry_mask%mark_zone(msh%wall)
384  call bdry_mask%mark_zones_from_list(msh%labeled_zones,&
385  'w', this%bc_labels)
386  call bdry_mask%finalize()
387  call bdry_mask%set_g(1.0_rp)
388  call bdry_mask%apply_scalar(this%bdry%x, this%dm_Xh%size())
389  call bdry_mask%free()
390 
391  call bdry_mask%init(this%dm_Xh)
392  call bdry_mask%mark_zone(msh%inlet)
393  call bdry_mask%mark_zones_from_list(msh%labeled_zones,&
394  'v', this%bc_labels)
395 
396  call bdry_mask%finalize()
397  call bdry_mask%set_g(2.0_rp)
398  call bdry_mask%apply_scalar(this%bdry%x, this%dm_Xh%size())
399  call bdry_mask%free()
400 
401  call bdry_mask%init(this%dm_Xh)
402  call bdry_mask%mark_zone(msh%outlet)
403  call bdry_mask%mark_zones_from_list(msh%labeled_zones,&
404  'o', this%bc_labels)
405  call bdry_mask%finalize()
406  call bdry_mask%set_g(3.0_rp)
407  call bdry_mask%apply_scalar(this%bdry%x, this%dm_Xh%size())
408  call bdry_mask%free()
409 
410  call bdry_mask%init(this%dm_Xh)
411  call bdry_mask%mark_zone(msh%sympln)
412  call bdry_mask%mark_zones_from_list(msh%labeled_zones,&
413  'sym', this%bc_labels)
414  call bdry_mask%finalize()
415  call bdry_mask%set_g(4.0_rp)
416  call bdry_mask%apply_scalar(this%bdry%x, this%dm_Xh%size())
417  call bdry_mask%free()
418 
419  call bdry_mask%init(this%dm_Xh)
420  call bdry_mask%mark_zone(msh%periodic)
421  call bdry_mask%finalize()
422  call bdry_mask%set_g(5.0_rp)
423  call bdry_mask%apply_scalar(this%bdry%x, this%dm_Xh%size())
424  call bdry_mask%free()
425 
426  call bdry_mask%init(this%dm_Xh)
427  call bdry_mask%mark_zone(msh%outlet_normal)
428  call bdry_mask%mark_zones_from_list(msh%labeled_zones,&
429  'on', this%bc_labels)
430  call bdry_mask%finalize()
431  call bdry_mask%set_g(6.0_rp)
432  call bdry_mask%apply_scalar(this%bdry%x, this%dm_Xh%size())
433  call bdry_mask%free()
434 
435  end if
436 
437  !
438  ! Setup right-hand side fields.
439  !
440  allocate(this%f_x)
441  allocate(this%f_y)
442  allocate(this%f_z)
443  call this%f_x%init(this%dm_Xh, fld_name="fluid_rhs_x")
444  call this%f_y%init(this%dm_Xh, fld_name="fluid_rhs_y")
445  call this%f_z%init(this%dm_Xh, fld_name="fluid_rhs_z")
446 
447  ! Initialize the source term
448  call this%source_term%init(params, this%f_x, this%f_y, this%f_z, this%c_Xh,&
449  user)
450 
451  end subroutine fluid_scheme_init_common

501  subroutine fluid_scheme_init_all(this, msh, lx, params, kspv_init, kspp_init,&
502  scheme, user, material_properties)
503  implicit none
504  class(fluid_scheme_t), target, intent(inout) :: this
505  type(mesh_t), target, intent(inout) :: msh
506  integer, intent(inout) :: lx
507  type(json_file), target, intent(inout) :: params
508  type(user_t), target, intent(in) :: user
509  type(material_properties_t), target, intent(inout) :: material_properties
510  logical :: kspv_init
511  logical :: kspp_init
512  character(len=*), intent(in) :: scheme
513  real(kind=rp) :: real_val, dong_delta, dong_uchar
514  character(len=:), allocatable :: string_val1, string_val2
515  integer :: integer_val
516 
517  call fluid_scheme_init_common(this, msh, lx, params, scheme, user, &
518  material_properties)
519 
520  call neko_field_registry%add_field(this%dm_Xh, 'u')
521  call neko_field_registry%add_field(this%dm_Xh, 'v')
522  call neko_field_registry%add_field(this%dm_Xh, 'w')
523  call neko_field_registry%add_field(this%dm_Xh, 'p')
524  this%u => neko_field_registry%get_field('u')
525  this%v => neko_field_registry%get_field('v')
526  this%w => neko_field_registry%get_field('w')
527  this%p => neko_field_registry%get_field('p')
528 
529  !
530  ! Setup pressure boundary conditions
531  !
532  call bc_list_init(this%bclst_prs)
533  call this%bc_prs%init(this%dm_Xh)
534  call this%bc_prs%mark_zones_from_list(msh%labeled_zones,&
535  'o', this%bc_labels)
536  call this%bc_prs%mark_zones_from_list(msh%labeled_zones,&
537  'on', this%bc_labels)
538 
539  if (msh%outlet%size .gt. 0) then
540  call this%bc_prs%mark_zone(msh%outlet)
541  end if
542  if (msh%outlet_normal%size .gt. 0) then
543  call this%bc_prs%mark_zone(msh%outlet_normal)
544  end if
545 
546  call this%bc_prs%finalize()
547  call this%bc_prs%set_g(0.0_rp)
548  call bc_list_add(this%bclst_prs, this%bc_prs)
549  call this%bc_dong%init(this%dm_Xh)
