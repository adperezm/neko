160  call neko_log%message(log_buf, neko_log_quiet)
161  write(log_buf, '(a)') 'Running on: '
162  sw = 10
163  if (pe_size .lt. 1e1) then
164  write(log_buf(13:), '(i1,a)') pe_size, ' MPI '
165  if (pe_size .eq. 1) then
166  write(log_buf(19:), '(a)') 'rank'
167  sw = 9
168  else
169  write(log_buf(19:), '(a)') 'ranks'
170  end if
171  rw = 1
172  else if (pe_size .lt. 1e2) then
173  write(log_buf(13:), '(i2,a)') pe_size, ' MPI ranks'
174  rw = 2
175  else if (pe_size .lt. 1e3) then
176  write(log_buf(13:), '(i3,a)') pe_size, ' MPI ranks'
177  rw = 3
178  else if (pe_size .lt. 1e4) then

181  else if (pe_size .lt. 1e5) then
182  write(log_buf(13:), '(i5,a)') pe_size, ' MPI ranks'
183  rw = 5
184  else

187  end if

195 
196  if (nthrds .gt. 1) then

199  nthrds, ' thrds each'
200  else if (nthrds .lt. 1e2) then
201  write(log_buf(13 + rw + sw:), '(a,i2,a)') ', using ', &
202  nthrds, ' thrds each'
203  else if (nthrds .lt. 1e3) then
204  write(log_buf(13 + rw + sw:), '(a,i3,a)') ', using ', &

206  else if (nthrds .lt. 1e4) then
207  write(log_buf(13 + rw + sw:), '(a,i4,a)') ', using ', &
208  nthrds, ' thrds each'
209  end if
210  end if
211  call neko_log%message(log_buf, neko_log_quiet)
212 
213  write(log_buf, '(a)') 'CPU type : '
214  call system_cpu_name(log_buf(13:))
215  call neko_log%message(log_buf, neko_log_quiet)
216 
217  write(log_buf, '(a)') 'Bcknd type: '
218  if (neko_bcknd_sx .eq. 1) then
219  write(log_buf(13:), '(a)') 'SX-Aurora'
220  else if (neko_bcknd_xsmm .eq. 1) then
221  write(log_buf(13:), '(a)') 'CPU (libxsmm)'
222  else if (neko_bcknd_cuda .eq. 1) then
223  write(log_buf(13:), '(a)') 'Accelerator (CUDA)'
224  else if (neko_bcknd_hip .eq. 1) then
225  write(log_buf(13:), '(a)') 'Accelerator (HIP)'
226  else if (neko_bcknd_opencl .eq. 1) then
227  write(log_buf(13:), '(a)') 'Accelerator (OpenCL)'
228  else
229  write(log_buf(13:), '(a)') 'CPU'
230  end if
231  call neko_log%message(log_buf, neko_log_quiet)
232 
233  if (neko_bcknd_hip .eq. 1 .or. neko_bcknd_cuda .eq. 1 .or. &
234  neko_bcknd_opencl .eq. 1) then
235  write(log_buf, '(a)') 'Dev. name : '
236  call device_name(log_buf(13:))
237  call neko_log%message(log_buf, neko_log_quiet)
238  end if
239 
240  write(log_buf, '(a)') 'Real type : '
241  select case (rp)
242  case (real32)
243  write(log_buf(13:), '(a)') 'single precision'
244  case (real64)
245  write(log_buf(13:), '(a)') 'double precision'
246  case (real128)
247  write(log_buf(13:), '(a)') 'quad precision'
248  end select
249  call neko_log%message(log_buf, neko_log_quiet)
250 
251  call neko_log%end()
252 
253  !
254  ! Create case
255  !
256  call case_init(c, case_file)
257 
258  !
259  ! Create simulation components
260  !
261  call neko_simcomps%init(c)
262 
263  end if
264 
265  end subroutine neko_init
