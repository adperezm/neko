402  subroutine mesh_generate_flags(this)
403  type(mesh_t), intent(inout) :: this
404  real(kind=dp) :: u(3), v(3), w(3), temp
405  integer :: e
406 
407  do e = 1, this%nelv
408  if (this%gdim .eq. 2) then
409  this%dfrmd_el(e) = .false.
410  u = this%elements(e)%e%pts(2)%p%x - this%elements(e)%e%pts(1)%p%x
411  v = this%elements(e)%e%pts(3)%p%x - this%elements(e)%e%pts(1)%p%x
412  temp = u(1)*v(1) + u(2)*v(2)
413  if(.not. abscmp(temp, 0d0)) this%dfrmd_el(e) = .true.
414  else
415  this%dfrmd_el(e) = .false.
416  u = this%elements(e)%e%pts(2)%p%x - this%elements(e)%e%pts(1)%p%x
417  v = this%elements(e)%e%pts(3)%p%x - this%elements(e)%e%pts(1)%p%x
418  w = this%elements(e)%e%pts(5)%p%x - this%elements(e)%e%pts(1)%p%x
419  temp = u(1)*v(1) + u(2)*v(2) + u(3)*v(3)
420  if(.not. abscmp(temp, 0d0)) this%dfrmd_el(e) = .true.
421  temp = u(1)*w(1) + u(2)*w(2) + u(3)*w(3)
422  if(.not. abscmp(temp, 0d0)) this%dfrmd_el(e) = .true.
423  u = this%elements(e)%e%pts(7)%p%x - this%elements(e)%e%pts(8)%p%x
424  v = this%elements(e)%e%pts(6)%p%x - this%elements(e)%e%pts(8)%p%x
425  w = this%elements(e)%e%pts(4)%p%x - this%elements(e)%e%pts(8)%p%x
426  temp = u(1)*v(1) + u(2)*v(2) + u(3)*v(3)
427  if(.not. abscmp(temp, 0d0)) this%dfrmd_el(e) = .true.
428  temp = u(1)*w(1) + u(2)*w(2) + u(3)*w(3)
429  if(.not. abscmp(temp, 0d0)) this%dfrmd_el(e) = .true.
430  end if
431  end do
432  end subroutine mesh_generate_flags

441  subroutine mesh_generate_conn(this)
442  class(mesh_t), target, intent(inout) :: this
443  type(tuple_i4_t) :: edge
444  type(tuple4_i4_t) :: face, face_comp
445  type(tuple_i4_t) :: facet_data
446  type(stack_i4_t) :: neigh_order
447 
448  integer :: i, j, k, ierr, el_glb_idx, n_sides, n_nodes, src, dst
449 
450  if (this%lconn) return
451 
452  if (.not. this%lgenc) return
453 
454  if (this%gdim .eq. 2) then
455  n_sides = 4
456  n_nodes = 2
457  else
458  n_sides = 6
459  n_nodes = 4
460  end if
461 
462  ! Compute global number of unique points
463  call mpi_allreduce(this%max_pts_id, this%glb_mpts, 1, &
464  mpi_integer, mpi_max, neko_comm, ierr)
465 
466  !
467  ! Find all (local) boundaries
468  !
469 
473  select type (fmp => this%facet_map)
474  type is(htable_i4t2_t)
475  do k = 1, 2
476  do i = 1, this%nelv
477  el_glb_idx = i + this%offset_el
478  do j = 1, n_sides
479  call this%elements(i)%e%facet_id(edge, j)
480 
481  ! Assume that all facets are on the exterior
482  facet_data%x = (/ 0, 0/)
483 
484  !check it this face has shown up earlier
485  if (fmp%get(edge, facet_data) .eq. 0) then
486  !if element is already recognized on face
487  if (facet_data%x(1) .eq. el_glb_idx ) then
488  this%facet_neigh(j, i) = facet_data%x(2)
489  else if( facet_data%x(2) .eq. el_glb_idx) then
490  this%facet_neigh(j, i) = facet_data%x(1)
491  !if this is the second element, arrange so low id is first
492  else if(facet_data%x(1) .gt. el_glb_idx) then
493  facet_data%x(2) = facet_data%x(1)
494  facet_data%x(1) = el_glb_idx
495  this%facet_neigh(j, i) = facet_data%x(2)
496  call fmp%set(edge, facet_data)
497  else if(facet_data%x(1) .lt. el_glb_idx) then
498  facet_data%x(2) = el_glb_idx
499  this%facet_neigh(j, i) = facet_data%x(1)
500  call fmp%set(edge, facet_data)
501  end if
502  else
503  facet_data%x(1) = el_glb_idx
504  this%facet_neigh(j, i) = facet_data%x(2)
505  call fmp%set(edge, facet_data)
506  end if
507  end do
508  end do
509  end do
510  type is(htable_i4t4_t)
511 
512  do k = 1, 2
513  do i = 1, this%nelv
514  el_glb_idx = i + this%offset_el
515  do j = 1, n_sides
516  call this%elements(i)%e%facet_id(face, j)
517 
518  facet_data%x = (/ 0, 0/)
519 
520  !check it this face has shown up earlier
521  if (fmp%get(face, facet_data) .eq. 0) then
522  !if element is already recognized on face
523  if (facet_data%x(1) .eq. el_glb_idx ) then
524  this%facet_neigh(j, i) = facet_data%x(2)
525  call this%elements(i)%e%facet_id(face_comp, &
526  j+(2*mod(j,2)-1))
527  if (face_comp .eq. face) then
528  facet_data%x(2) = el_glb_idx
529  this%facet_neigh(j, i) = facet_data%x(1)
530  call fmp%set(face, facet_data)
531  end if
532  else if( facet_data%x(2) .eq. el_glb_idx) then
533  this%facet_neigh(j, i) = facet_data%x(1)
534  !if this is the second element, arrange so low id is first
535  else if(facet_data%x(1) .gt. el_glb_idx) then
536  facet_data%x(2) = facet_data%x(1)
537  facet_data%x(1) = el_glb_idx
538  this%facet_neigh(j, i) = facet_data%x(2)
539  call fmp%set(face, facet_data)
540  else if(facet_data%x(1) .lt. el_glb_idx) then
541  facet_data%x(2) = el_glb_idx
542  this%facet_neigh(j, i) = facet_data%x(1)
543  call fmp%set(face, facet_data)
544  end if
545  else
546  facet_data%x(1) = el_glb_idx
547  this%facet_neigh(j, i) = 0
548  call fmp%set(face, facet_data)
549  end if
550  end do
551  end do
552  end do
553  class default
554  call neko_error('Invalid facet map')
555  end select
556 
557 
558  !
559  ! Find all external (between PEs) boundaries
560  !
561  if (pe_size .gt. 1) then
562 
563  call mesh_generate_external_point_conn(this)
564 
565  !
566  ! Generate neighbour exchange order
567  !
568  call neigh_order%init(pe_size)
569 
570  do i = 1, pe_size - 1
571  src = modulo(pe_rank - i + pe_size, pe_size)
572  dst = modulo(pe_rank + i, pe_size)
573  if (this%neigh(src) .or. this%neigh(dst)) then
574  j = i ! adhere to standards...
575  call neigh_order%push(j)
576  end if
577  end do
578 
579  allocate(this%neigh_order(neigh_order%size()))
580  select type(order => neigh_order%data)
581  type is (integer)
582  do i = 1, neigh_order%size()
583  this%neigh_order(i) = order(i)
584  end do
585  end select
586  call neigh_order%free()
587 
588  call mesh_generate_external_facet_conn(this)
589  else
590  allocate(this%neigh_order(1))
591  this%neigh_order = 1
592  end if
593 
594  !
595  ! Find all internal/extenral edge connections
596  ! (Note it needs to be called after external point connections has
597  ! been established)
598  !
599  if (this%gdim .eq. 3) then
600  call mesh_generate_edge_conn(this)
601  end if
602 
603 
604  call mesh_generate_facet_numbering(this)
605 
606  this%lconn = .true.
607 
608  end subroutine mesh_generate_conn
