365  real hip_vlsc3(void *u, void *v, void *w, int *n) {
366 
367  const dim3 nthrds(1024, 1, 1);
368  const dim3 nblcks(((*n)+1024 - 1)/ 1024, 1, 1);
369  const int nb = ((*n) + 1024 - 1)/ 1024;
370  const hipStream_t stream = (hipStream_t) glb_cmd_queue;
371 
372  if ( nb > red_s){
373  red_s = nb;
374  if (bufred != NULL) {
375  HIP_CHECK(hipHostFree(bufred));
376  HIP_CHECK(hipFree(bufred_d));
377  }
378  HIP_CHECK(hipHostMalloc(&bufred,nb*sizeof(real),hipHostMallocDefault));
379  HIP_CHECK(hipMalloc(&bufred_d, nb*sizeof(real)));
380  }
381 
382  hipLaunchKernelGGL(HIP_KERNEL_NAME(glsc3_kernel<real>),
383  nblcks, nthrds, 0, stream,
384  (real *) u, (real *) v,
385  (real *) w, bufred_d, *n);
386  HIP_CHECK(hipGetLastError());
387  hipLaunchKernelGGL(HIP_KERNEL_NAME(reduce_kernel<real>),
388  1, 1024, 0, stream, bufred_d, nb);
389  HIP_CHECK(hipGetLastError());
390 
391  HIP_CHECK(hipMemcpyAsync(bufred, bufred_d, sizeof(real),
392  hipMemcpyDeviceToHost, stream));
393  hipStreamSynchronize(stream);
394 
395  return bufred[0];
