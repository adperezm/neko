55  subroutine point_zone_factory(point_zone, json, dof)
56  class(point_zone_t), allocatable, intent(inout) :: point_zone
57  type(json_file), intent(inout) :: json
58  type(dofmap_t), intent(inout) :: dof
59  character(len=:), allocatable :: zone_type
60 
61  call json_get(json, "geometry", zone_type)
62 
63  if (trim(zone_type) .eq. "box") then
64  allocate(box_point_zone_t::point_zone)
65  else if (trim(zone_type) .eq. "sphere") then
66  allocate(sphere_point_zone_t::point_zone)
67  else if (trim(zone_type) .eq. "cylinder") then
68  allocate(cylinder_point_zone_t::point_zone)
69  else
70  call neko_error("Unknown source term "//trim(zone_type)//"! Valid &
71  &source terms are 'box', 'sphere', 'cylinder'.")
72  end if
73 
74  call point_zone%init(json, dof%size())
75 
76  call point_zone%map(dof)
77  call point_zone%finalize()
78 
79  end subroutine point_zone_factory
