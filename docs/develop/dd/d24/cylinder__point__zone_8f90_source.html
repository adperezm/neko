170  real(kind=rp) :: distance
171 
172  p = [x, y, z]
173 
174  line = this%p1 - this%p0
175  t = dot_product(p - this%p0, line) / norm2(line)
176 
177  projection = this%p0 + t * line
178  distance = norm2(projection - p)
179 
180  is_inside = t >= 0.0_rp .and. t <= 1.0_rp .and. distance <= this%radius
181
