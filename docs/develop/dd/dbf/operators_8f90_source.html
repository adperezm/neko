303  type(coef_t), intent(in) :: coef
304  real(kind=rp), intent(inout) :: s11(u%Xh%lx, u%Xh%ly, u%Xh%lz, u%msh%nelv)
305  real(kind=rp), intent(inout) :: s22(u%Xh%lx, u%Xh%ly, u%Xh%lz, u%msh%nelv)
306  real(kind=rp), intent(inout) :: s33(u%Xh%lx, u%Xh%ly, u%Xh%lz, u%msh%nelv)
307  real(kind=rp), intent(inout) :: s12(u%Xh%lx, u%Xh%ly, u%Xh%lz, u%msh%nelv)
308  real(kind=rp), intent(inout) :: s23(u%Xh%lx, u%Xh%ly, u%Xh%lz, u%msh%nelv)

374  subroutine lambda2op(lambda2, u, v, w, coef)
375  type(coef_t), intent(in) :: coef
376  type(field_t), intent(inout) :: lambda2
377  type(field_t), intent(in) :: u, v, w
378  real(kind=rp) :: grad(coef%Xh%lxyz,3,3)
379  integer :: temp_indices(9), e, i, ind_sort(3)
380  real(kind=rp) :: eigen(3), b, c, d, q, r, theta, l2
381  real(kind=rp) :: s11, s22, s33, s12, s13, s23, o12, o13, o23
382  real(kind=rp) :: a11, a22, a33, a12, a13, a23
383 
384  if (neko_bcknd_device .eq. 1) then
385  call opr_device_lambda2(lambda2, u, v, w, coef)
386  else
387  do e = 1, coef%msh%nelv
388  call opgrad(grad(1,1,1), grad(1,1,2), grad(1,1,3), &
389  u%x(1,1,1,e),coef,e,e)
390  call opgrad(grad(1,2,1), grad(1,2,2), grad(1,2,3), &
391  v%x(1,1,1,e),coef,e,e)
392  call opgrad(grad(1,3,1), grad(1,3,2), grad(1,3,3), &
393  w%x(1,1,1,e),coef,e,e)
394 
395  do i = 1, coef%Xh%lxyz
396  s11 = grad(i,1,1)
397  s22 = grad(i,2,2)
398  s33 = grad(i,3,3)
399 
400 
401  s12 = 0.5*(grad(i,1,2) + grad(i,2,1))
402  s13 = 0.5*(grad(i,1,3) + grad(i,3,1))
403  s23 = 0.5*(grad(i,2,3) + grad(i,3,2))
404 
405  o12 = 0.5*(grad(i,1,2) - grad(i,2,1))
406  o13 = 0.5*(grad(i,1,3) - grad(i,3,1))
407  o23 = 0.5*(grad(i,2,3) - grad(i,3,2))
408 
409  a11 = s11*s11 + s12*s12 + s13*s13 - o12*o12 - o13*o13
410  a12 = s11 * s12 + s12 * s22 + s13 * s23 - o13 * o23
411  a13 = s11 * s13 + s12 * s23 + s13 * s33 + o12 * o23
412 
413  a22 = s12*s12 + s22*s22 + s23*s23 - o12*o12 - o23*o23
414  a23 = s12 * s13 + s22 * s23 + s23 * s33 - o12 * o13
415  a33 = s13*s13 + s23*s23 + s33*s33 - o13*o13 - o23*o23
416 
417 
418  b = -(a11 + a22 + a33)
419  c = -(a12*a12 + a13*a13 + a23*a23 &
420  - a11 * a22 - a11 * a33 - a22 * a33)
421  d = -(2.0 * a12 * a13 * a23 - a11 * a23*a23 &
422  - a22 * a13*a13 - a33 * a12*a12 + a11 * a22 * a33)
423 
424 
425  q = (3.0 * c - b*b) / 9.0
426  r = (9.0 * c * b - 27.0 * d - 2.0 * b*b*b) / 54.0
427  theta = acos( r / sqrt(-q*q*q) )
428 
429  eigen(1) = 2.0 * sqrt(-q) * cos(theta / 3.0) - b / 3.0
430  eigen(2) = 2.0 * sqrt(-q) * cos((theta + 2.0 * pi) / 3.0) - b / 3.0
431  eigen(3) = 2.0 * sqrt(-q) * cos((theta + 4.0 * pi) / 3.0) - b / 3.0
432 
433  if (eigen(1) .le. eigen(2) .and. eigen(2) .le. eigen(3)) then
434  l2 = eigen(2)
435  else if (eigen(3) .le. eigen(2) .and. eigen(2) .le. eigen(1)) then
436  l2 = eigen(2)
437  else if (eigen(1) .le. eigen(3) .and. eigen(3) .le. eigen(2)) then
438  l2 = eigen(3)
439  else if (eigen(2) .le. eigen(3) .and. eigen(3) .le. eigen(1)) then
440  l2 = eigen(3)
441  else if (eigen(2) .le. eigen(1) .and. eigen(1) .le. eigen(3)) then
442  l2 = eigen(1)
443  else if (eigen(3) .le. eigen(1) .and. eigen(1) .le. eigen(2)) then
444  l2 = eigen(1)
445  else
446  l2 = 0.0
447  end if
448  lambda2%x(i,1,1,e) = l2/(coef%B(i,1,1,e)**2)
449  end do
450  end do
451  end if
452 
453  end subroutine lambda2op
